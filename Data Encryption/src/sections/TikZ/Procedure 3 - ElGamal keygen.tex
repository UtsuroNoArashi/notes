    \begin{procedure}
        \caption{ElGamal-Keygen()}\label{proc:ElGamal-keygen}
            \nl Select a multiplicative group $ \zset_{p}^{}*$ generated by an element $\gamma$
                \DontPrintSemicolon\;
            \nl Randomly choose an integer $a,  1 \le a \le p - 2$ and compute $\gamma^{a} \mod p$
                \DontPrintSemicolon\; 
            \nl Use the triplet $(p, \gamma, \gamma^{a})$ as public-key, and $a$ as the private one.
    \end{procedure}
